#!/usr/bin/env bash
# tests/mc/SLICE-MC6-test.sh
# Run after SLICE MC6 lands. Verifies:
#   - Controllers created via map_creator get auto-assigned controllerKey (A, B, ...)
#   - and controllerColor from the DEFAULT_COLORS rotation
#   - Zone creation requires + carries parent controller fields
#   - generateZoneLabel Jest test passes

set -uo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
source "$SCRIPT_DIR/_setup.sh"

require_env API_BASE_URL DATABASE_URL ADMIN_USERNAME ADMIN_PASSWORD

echo "SLICE MC6 — Controller-Zone Hierarchy tests"

section "setup"

login_as "$ADMIN_USERNAME" "$ADMIN_PASSWORD" >/dev/null
pass "admin login"

MC_USERNAME="__mc6_mc_$RUN_SUFFIX"
TEST_PASS="testpass123"
api_status POST /api/admin/users "{\"username\":\"$MC_USERNAME\",\"password\":\"$TEST_PASS\",\"displayName\":\"x\",\"role\":\"map_creator\"}" >/dev/null
MC_ID=$(psql_value "SELECT id FROM users WHERE username = '$MC_USERNAME'")

TEST_COMM=$(api_body POST /api/communities "{\"name\":\"__mc6_comm_$RUN_SUFFIX\"}")
TEST_COMM_ID=$(echo "$TEST_COMM" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id',''))" 2>/dev/null)
psql "$DATABASE_URL" -c "INSERT INTO community_members (community_id, user_id) VALUES ('$TEST_COMM_ID', '$MC_ID')" >/dev/null

pass "test community + map_creator member set up ($TEST_COMM_ID)"

logout

section "Controller auto-key + auto-color (first three)"

login_as "$MC_USERNAME" "$TEST_PASS" >/dev/null

create_controller() {
  local feat="$1"
  api_body POST /api/assets "{\"communityId\":\"$TEST_COMM_ID\",\"assetType\":\"controller\",\"label\":\"$feat\",\"latitude\":39.7,\"longitude\":-104.9,\"geometryType\":\"Point\",\"featureRef\":\"$feat\"}"
}

C1_BODY=$(create_controller "mc6-ctrl-1-$RUN_SUFFIX")
C2_BODY=$(create_controller "mc6-ctrl-2-$RUN_SUFFIX")
C3_BODY=$(create_controller "mc6-ctrl-3-$RUN_SUFFIX")

C1_ID=$(echo "$C1_BODY" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id',''))" 2>/dev/null)
C2_ID=$(echo "$C2_BODY" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id',''))" 2>/dev/null)
C3_ID=$(echo "$C3_BODY" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id',''))" 2>/dev/null)

# Look up the controllerKey and controllerColor from asset_properties
get_prop() {
  local asset_id="$1"
  local key="$2"
  psql_value "SELECT value FROM asset_properties WHERE asset_id = '$asset_id' AND key = '$key'"
}

C1_KEY=$(get_prop "$C1_ID" "controllerKey")
C2_KEY=$(get_prop "$C2_ID" "controllerKey")
C3_KEY=$(get_prop "$C3_ID" "controllerKey")

if [ "$C1_KEY" = "A" ]; then
  pass "Controller 1 auto-assigned controllerKey='A'"
else
  fail "Controller 1 controllerKey is '$C1_KEY' (expected 'A')"
fi

if [ "$C2_KEY" = "B" ]; then
  pass "Controller 2 auto-assigned controllerKey='B'"
else
  fail "Controller 2 controllerKey is '$C2_KEY' (expected 'B')"
fi

if [ "$C3_KEY" = "C" ]; then
  pass "Controller 3 auto-assigned controllerKey='C'"
else
  fail "Controller 3 controllerKey is '$C3_KEY' (expected 'C')"
fi

C1_COLOR=$(get_prop "$C1_ID" "controllerColor")
C2_COLOR=$(get_prop "$C2_ID" "controllerColor")

if [ -n "$C1_COLOR" ] && [[ "$C1_COLOR" =~ ^#[0-9a-fA-F]{6}$ ]]; then
  pass "Controller 1 has controllerColor=$C1_COLOR"
else
  fail "Controller 1 controllerColor missing or malformed: '$C1_COLOR'"
fi

if [ -n "$C2_COLOR" ] && [ "$C2_COLOR" != "$C1_COLOR" ]; then
  pass "Controller 2 controllerColor differs from Controller 1"
else
  fail "Controller 2 controllerColor matches Controller 1 ('$C2_COLOR') — palette rotation broken"
fi

section "Zone creation with parent controller"

ZONE_FEAT="mc6-zone-1-$RUN_SUFFIX"
ZONE_BODY=$(api_body POST /api/assets "{
  \"communityId\":\"$TEST_COMM_ID\",
  \"assetType\":\"zone\",
  \"label\":\"Zone 1\",
  \"latitude\":39.7001,
  \"longitude\":-104.9001,
  \"geometryType\":\"Point\",
  \"featureRef\":\"$ZONE_FEAT\",
  \"properties\":{
    \"controllerFeatureRef\":\"mc6-ctrl-1-$RUN_SUFFIX\",
    \"controllerLabel\":\"mc6-ctrl-1-$RUN_SUFFIX\",
    \"controllerColor\":\"$C1_COLOR\",
    \"zoneNumber\":\"1\"
  }
}")
ZONE_ID=$(echo "$ZONE_BODY" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id',''))" 2>/dev/null)

if [ -n "$ZONE_ID" ]; then
  pass "Zone with parent Controller created (id: $ZONE_ID)"
else
  fail "Zone creation failed: $ZONE_BODY"
fi

ZONE_PARENT_REF=$(get_prop "$ZONE_ID" "controllerFeatureRef")
ZONE_NUMBER=$(get_prop "$ZONE_ID" "zoneNumber")
ZONE_COLOR=$(get_prop "$ZONE_ID" "controllerColor")

if [ "$ZONE_PARENT_REF" = "mc6-ctrl-1-$RUN_SUFFIX" ]; then
  pass "Zone properties.controllerFeatureRef points to parent Controller"
else
  fail "Zone controllerFeatureRef is '$ZONE_PARENT_REF'"
fi

if [ "$ZONE_NUMBER" = "1" ]; then
  pass "Zone properties.zoneNumber = 1"
else
  fail "Zone zoneNumber is '$ZONE_NUMBER' (expected '1')"
fi

if [ "$ZONE_COLOR" = "$C1_COLOR" ]; then
  pass "Zone inherits controllerColor from parent"
else
  fail "Zone controllerColor is '$ZONE_COLOR' (expected '$C1_COLOR')"
fi

section "GET /api/communities/:id/controllers returns the new tree"

CONTROLLERS_BODY=$(api_body GET "/api/communities/$TEST_COMM_ID/controllers")
CTRL_COUNT=$(echo "$CONTROLLERS_BODY" | python3 -c "import json,sys; print(len(json.load(sys.stdin)))" 2>/dev/null)
if [ "$CTRL_COUNT" = "3" ]; then
  pass "endpoint returns 3 controllers"
else
  fail "endpoint returned $CTRL_COUNT controllers (expected 3)"
fi

logout

section "Jest: generateZoneLabel logic"

if [ -f "artifacts/vrtsync-mobile/__tests__/mc/generateZoneLabel.test.ts" ]; then
  pnpm --filter @workspace/vrtsync-mobile run test -- __tests__/mc/generateZoneLabel.test.ts > /tmp/mc6-jest.log 2>&1
  if grep -q "Tests:.*passed" /tmp/mc6-jest.log; then
    pass "generateZoneLabel.test.ts passed"
  else
    fail "generateZoneLabel.test.ts failed — see /tmp/mc6-jest.log"
    tail -10 /tmp/mc6-jest.log
  fi
else
  note "no Jest test at __tests__/mc/generateZoneLabel.test.ts"
fi

section "cleanup"

login_as "$ADMIN_USERNAME" "$ADMIN_PASSWORD" >/dev/null
psql "$DATABASE_URL" -c "DELETE FROM asset_properties WHERE asset_id IN (SELECT id FROM assets WHERE community_id = '$TEST_COMM_ID')" >/dev/null 2>&1
psql "$DATABASE_URL" -c "DELETE FROM assets WHERE community_id = '$TEST_COMM_ID'" >/dev/null 2>&1
psql "$DATABASE_URL" -c "DELETE FROM community_members WHERE community_id = '$TEST_COMM_ID'" >/dev/null 2>&1
psql "$DATABASE_URL" -c "DELETE FROM communities WHERE id = '$TEST_COMM_ID'" >/dev/null 2>&1
psql "$DATABASE_URL" -c "DELETE FROM users WHERE username = '$MC_USERNAME'" >/dev/null 2>&1
pass "cleanup complete"
logout

summarize
